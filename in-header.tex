% Redefine the hyperref command to underline the link text
\let\oldhyperref\hyperref
\renewcommand{\hyperref}[2][\empty]{%
  \ifx\empty#1\empty
    \oldhyperref{#2}{\underline{#2}}% If no label is passed, just underline the link text
  \else
    \oldhyperref[#1]{\underline{#2}}% If a label is passed, keep it and underline the link text
  \fi
}
