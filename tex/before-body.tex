% Suppress the \chapter{Impressum} Quarto generates for imprint.qmd.
% Starred form (\chapter*) is passed through so \tableofcontents can render
% its heading (e.g. "Inhaltsverzeichnis"). Only the first non-starred call
% (the Impressum chapter) is gobbled; then \chapter is fully restored.
\makeatletter
\let\quartoOrigChapter\chapter
\newcommand{\quartoChapterStarHelper}[1]{\quartoOrigChapter*{#1}}
\renewcommand{\chapter}{\@ifstar
  {\quartoChapterStarHelper}%
  {\global\let\chapter\quartoOrigChapter\quartoGobbleChapOpt}%
}
\def\quartoGobbleChapOpt{\@ifnextchar[{\quartoGobbleChapTwo}{\quartoGobbleChapOne}}
\def\quartoGobbleChapTwo[#1]#2{}
\def\quartoGobbleChapOne#1{}
\makeatother
